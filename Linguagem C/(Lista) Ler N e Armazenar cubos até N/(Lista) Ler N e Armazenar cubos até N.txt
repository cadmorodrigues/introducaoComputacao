/* Programa que lê um número e calcula os cubos de um até esse determinado número N */
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

/* definicao de tipo */
struct lista
{
    int info;
    struct lista* prox;
};
typedef struct lista Lista;

/* definicao de funcoes */
Lista* lst_cria(void)
{ Lista* n = (Lista*)malloc(sizeof(Lista));
  return n = NULL;
}
Lista* lst_insere(Lista* l, int i)
{
  Lista* novo  = (Lista*)malloc(sizeof(Lista));
  novo -> info = i;
  novo -> prox = l;
  return novo;
}

void lst_imprime(Lista* l)
{
    Lista* p;
    for (p = l;p != NULL;p = p->prox)
      printf("%d\t",p->info);
    printf("\n");
}


void lst_libera(Lista* l)
{
    Lista* p = l;
    while (p != NULL)
    {
        Lista* t = p->prox;
        free(p);
        p = t;
    }
}

int main()
{
    Lista* l;
    int n,i,numero;
    printf("Digite um numero inteiro : ");
    scanf("%d",&n);
    l = lst_cria();
    for(i = 1;i<= n;i++)
    {
        numero=pow(i,3);
        l = lst_insere(l,numero);

    }
    lst_imprime(l);
    printf("\n");
    lst_libera(l);
    return 0;
}
